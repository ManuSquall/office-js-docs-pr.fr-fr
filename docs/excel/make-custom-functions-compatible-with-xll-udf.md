```xml
  ...
  <EquivalentAddins>
    <EquivalentAddin>
      <ProgId>ContosoCOMAddin</ProgId>
      <Type>COM</Type>
    </EquivalentAddin>

    <EquivalentAddin>

    </EquivalentAddin>
  </EquivalentAddins>
</VersionOverrides>
```
